   169тЖТ   169тЖТfunction formatBytes(bytes: number): string {
   170тЖТ   170тЖТ  if (bytes === 0) return '0 B';
   171тЖТ   171тЖТ  const k = 1024;
   172тЖТ   172тЖТ  const sizes = ['B', 'KB', 'MB', 'GB', 'TB'];
   173тЖТ   173тЖТ  const i = Math.floor(Math.log(bytes) / Math.log(k));
   174тЖТ   174тЖТ  return parseFloat((bytes / Math.pow(k, i)).toFixed(1)) + ' ' + sizes[i];
   175тЖТ   175тЖТ}
