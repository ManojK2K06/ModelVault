    11→// Realistic-looking fake SHA-256 hashes
    12→const fakeSha256 = (seed: number) =>
    13→  'a'.repeat(8) +
    14→  seed.toString(16).padStart(2, '0') +
    15→  'b'.repeat(8) +
    16→  (seed * 7).toString(16).padStart(4, '0') +
    17→  'c'.repeat(8) +
    18→  (seed * 13).toString(16).padStart(4, '0') +
    19→  'd'.repeat(8) +
    20→  (seed * 3).toString(16).padStart(4, '0') +
    21→  'e'.repeat(8) +
    22→  (seed * 11).toString(16).padStart(4, '0') +
    23→  'f'.repeat(16);
    24→
    25→const padHex = (n: number, len: number) => n.toString(16).padStart(len, '0');
    26→
    27→const sha256 = (seed: number) => {
    28→  // Produce a deterministic 64-char hex string
    29→  const s = padHex(seed, 64);
    30→  return (
    31→    'a3f7b2c1e9d40856f6a3b7c9d2e1f4a8b5c7d9e1f3a6b8c0d2e4f7a9b1c3d5e' +
    32→    s
    33→  ).slice(0, 64);
    34→};
